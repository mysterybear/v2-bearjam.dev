```js
import React, { useEffect } from 'react';
import WebMidi from "webmidi";

function App() {
  useEffect(() => {
    WebMidi.enable(err => {
      if (err) throw err

      const output = WebMidi.getOutputByName("Midi Through Port-0")
      if (!output) {
        throw Error("Midi Through Port-0 not found")
      }

      const draw = () => {
        output.playNote(`C${Math.floor(Math.random() * 6)}`)
        requestAnimationFrame(draw)
      }
      draw()
    })
  }, [])
  return (
    <p>foo</p>
  );
}

export default App;
```
